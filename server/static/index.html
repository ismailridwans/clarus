Example — Python (OpenAI Client)
from openai import OpenAI
import httpx
 
client = OpenAI(
    base_url=API_BASE_URL,
    api_key=HF_TOKEN
)
 
# Reset a deductive_liability episode
obs = httpx.post(f"{ENV_URL}/reset",
    json={"task_name": "deductive_liability", "seed": 1101}).json()
 
# Authenticate patient (required first step)
result = httpx.post(f"{ENV_URL}/step",
    json={"action": {"action_type": "authenticate_patient", "parameters": {}}}).json()
 
print(result["reward"]) # → 0.05
